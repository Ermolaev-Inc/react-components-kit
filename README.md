## Installation
React Components Kit is available as an [npm package](https://www.npmjs.com/package/erck)

```
npm i erck --save
```

## Usage
``` tsx
import React from "react";
import { InputOpen } from "erck";

export const Home: React.FC = () => {
  return (
    <InputOpen placeholder="login" />
  );  
};
```
